```cpp
class Solution {
public:
vector<vector<int>> result;
vector<int> path;
void helper(int &n,int& k,int startIndex){
    if(path.size()==k){
        result.emplace_back(path);
        return;
    }
    for(int i=startIndex;i<=n-(k-path.size())+1;i++){
        path.emplace_back(i);
        helper(n,k,i+1);
        path.pop_back();
    }
}
    vector<vector<int>> combine(int n, int k) {
        helper(n,k,1);
        return result;
    }
};
```

```cpp
        for(int i=start,j=end;i<j;i++,j--){

                return false;

            }else{
                continue;
            }
            backtracking(s, i+1);
            path.pop_back();
        }
    }

        backtracking(s,0);
```

所有这些机票都属于一个从

<details><summary>Click to expand</summary>

```cpp
class Solution {
    //出发机场->到达机场->航班次数
    std::unordered_map<string,map<string,int>> targets;

public:
    bool backtracking(int ticketNum,vector<string>& result){
        if(result.size()==ticketNum+1){
            return true;
        }
        for(auto& target:targets[result.back()]){
            if(target.second>0){//说明有达到的路线
                result.push_back(target.first);
                target.second--;
                if(backtracking(ticketNum,result)){
                    return true;
                }
                result.pop_back();
                target.second++;
            }
        }
        return false;
    }
    vector<string> findItinerary(vector<vector<string>>& tickets) {
        vector<string> result;
        for(const auto& vec: tickets){
            targets[vec[0]][vec[1]]++;
        }
        result.push_back("JFK");
        backtracking(tickets.size(),result);
        return result;
    }
};
```

每一种解法包含一个不同的 n 皇后问题 的棋子放置方案，该方案中 \'Q\' 和

```cpp
class Solution {

    bool is_valid(int row,int col,vector<string>& chessboard,int n){
        int count=0;
        for(int i=0;i<row;i++){
            if(chessboard[i][col]=='Q'){return false;}
        }
        for(int i=row-1,j=col-1;i>=0&&j>=0;i--,j--){
            if(chessboard[i][j]=='Q') return false;
        }

        }
```

```cpp
public:
    int openLock(vector<string>& deadends, string target) {
        if (target == "0000")
            return 0;
        auto prev_char = [](char c) -> char { return c == '0' ? '9' : c - 1; };
        auto next_char = [](char c) -> char { return c == '9' ? '0' : c + 1; };
        auto next_status = [&](string& status) -> array<string, 8> {

                char num = status[i];
                status[i] = prev_char(num);

                status[i] = next_char(num);
                next_stat[i * 2 + 1] = status;
                status[i] = num;
            }
            return next_stat;
        };

            return -1;
        }

                    if (next_stat == target) {
                        return step + 1;
                    }
                    q.emplace(next_stat, step + 1);
                    seen.insert(move(next_stat));
                }
            }
        }
```

```cpp
        dfs(n,0,0,0,0);
        return res;

            res++;
            return;

        while(bits>0){
            int pick=bits&-bits;

        }
```

```cpp
    static constexpr int SIDE=3;
public:
    int numberOfPatterns(int m, int n) {

        for(int i=m;i<=n;i++){
            set<int> s;
            count+=backtracking(i,s,0,0);
        }
        return count;

        if(index==patternLen){
            return 1;
        }
        int count=0;
        for(int i=1;i<=MAX_NUMBER;i++){
            if(!visited.contains(i) && isValid(visited,prev,i)){
                visited.insert(i);
                count+=backtracking(patternLen,visited,index+1,i);
                visited.erase(i);
            }
        }
        return count;

        if(prev==0){
            return true;
        }

        int prevCol=(prev-1)%SIDE+1;
        int currRow=(curr-1)/SIDE+1;
        int currCol=(curr-1)%SIDE+1;
        int midRow=(prevRow+currRow)/2;
        int midCol=(prevCol+currCol)/2;
        if(midRow*2==prevRow+currRow && midCol*2==prevCol+currCol){
            int midNum=(midRow-1)*SIDE+midCol;
            return visited.contains(midNum);
        }else{
            return true;
        }
```

```cpp
class Solution {
    unordered_set<string> dict,curLevel,nextLevel;
    // 反向图
    unordered_map<string,vector<string>> graph;
    // 记录路径，当生成一条有效路的时候，拷贝进ans
    deque<string> path;
    vector<vector<string>> ans;
    void build(vector<string>& wordList){
        dict=unordered_set<string>(wordList.begin(),wordList.end());
        graph.clear();
        ans.clear();
        curLevel.clear();
        nextLevel.clear();
    }
    // 回溯生成路径
    void dfs(string word,string aim){
        path.push_front(word);
        if(word==aim){
            ans.push_back(vector<string>(path.begin(),path.end()));
        }else if(graph.find(word)!=graph.end()){
            for(auto& next:graph[word]){
                dfs(next,aim);
            }
        }
        path.pop_front();
    }
    // begin->end, 一层层bfs建图，返回值：能找到就返回true,否则为false
    bool bfs(string begin,string end){
        bool find=false;
        curLevel.insert(begin);
        while(!curLevel.empty()){
            //移除当前层的单词
            for(auto& word:curLevel){
                dict.erase(word);
            }
            for(auto& word:curLevel){
                string w=word;
                for(int i=0;i<w.size();i++){
                    char old=w[i];
                    for(char ch='a';ch<='z';ch++){
                        w[i]=ch;
                        if(dict.find(w)!=dict.end() && w!=word){
                            if(w==end){
                                find=true;
                            }
                            graph[w].push_back(word);
                            nextLevel.insert(w);
                        }
                    }
                    w[i]=old;
                }
            }
            if(find){
                return true;
            }else{
                swap(curLevel,nextLevel);
                nextLevel.clear();
            }
        }
        return false;
    }
public:
    vector<vector<string>> findLadders(string beginWord, string endWord, vector<string>& wordList) {
        build(wordList);
        if(dict.find(endWord)==dict.end()){
            return ans;
        }
        if(bfs(beginWord,endWord)){
            dfs(endWord,beginWord);
        }
        return ans;
    }
```

```cpp
    struct TrieNode{
        string word;
        unordered_map<char,TrieNode*> children;
        TrieNode(){
            this->word="";
        }
    };
    void insertTrie(TrieNode *root,const string& word){
        auto node=root;
        for(auto c:word){
            if(!node->children.count(c)){
                node->children[c]=new TrieNode();
            }
            node=node->children[c];
        }
        node->word=word;
    }
```

# 267 回文排列Ⅱ

给定一个字符串 s ，返回 其重新排列组合后可能构成的所有回文字符串，并去除重复的组合 。

```cpp
class Solution {
public:
    vector<string> generatePalindromes(string s) {
        unordered_map<char,int> char_freq;
        for(auto ch:s){
            char_freq[ch]++;
        }
        int odd_cnt=0;
        char odd_char;
        for(auto& [c,freq]:char_freq){
            if(freq&1){
                odd_cnt++;
                odd_char=c;
                if(odd_cnt>1){
                    return {};
                }
            }
        }
        string mid="";
        if(odd_cnt){
            mid+=odd_char;
            char_freq[odd_char]--;
        }
        vector<string> ans;
        auto dfs=[&](this auto&& dfs,string cur,int n){
            if(cur.size()>n){
                return 0;
            }
            if(cur.size()==n){
                ans.push_back(cur);
            }
            for(auto &[c,freq]:char_freq){
                if(freq>0){
                    char_freq[c]-=2;
                    dfs(c+cur+c,n);
                    char_freq[c]+=2;
                }
            }
            return 0;
        };
        dfs(mid,s.size());
        return ans;
    }
```

# 291 单词规律Ⅱ

给你一种规律 pattern 和一个字符串 s，请你判断 s 是否和 pattern 的规律相匹配。

如果存在单个字符到 非空 字符串的 双射映射 ，那么字符串 s 匹配 pattern ，即：如果 pattern 中的每个字符都被它映射到的字符串替换，那么最终的字符串则为 s 。双射 意味着映射双方一一对应，不会存在两个字符映射到同一个字符串，也不会存在一个字符分别映射到两个不同的字符串。

<details>

```cpp
class Solution {
    unordered_map<char,string> map;
    bool existsInMap(string_view& value){
        for(auto& [_,v]:map){
            if(v==value) return true;
        }
        return false;
    }
public:
    bool wordPatternMatch(string_view pattern, string_view s) {
        if(pattern=="") return s=="";
        char cur=pattern[0];
        for(int i=1;i<=s.size()-pattern.size()+1;i++){
            auto curstr=s.substr(0,i);
            auto it=map.find(cur);
            if((it!=map.end()&& it->second==curstr)||(it==map.end()&&!existsInMap(curstr))){
                map[cur]=curstr;
                if(wordPatternMatch(pattern.substr(1),s.substr(curstr.size()))){
                    return true;
                }
                if(it==map.end()){
                    map.erase(cur);
                }
            }
        }
        return false;
    }
};
```

```cpp
                res.push_back(str);
```

# 320 列举单词的全部缩写

单词的 广义缩写词 可以通过下述步骤构造：先取任意数量的 不重叠、不相邻 的

# 679 24点游戏

给定一个长度为4的整数数组 cards 。你有 4 张卡片，每张卡片上都包含一个范围在 [1,9] 的数字。您应该使用运算符 ['+', '-', '*', '/'] 和括号 '(' 和 ')' 将这些卡片上的数字排列成数学表达式，以获得值24。

你须遵守以下规则:

除法运算符 '/' 表示实数除法，而不是整数除法。
例如， 4 /(1 - 2 / 3)= 4 /(1 / 3)= 12 。
每个运算都在两个数字之间。特别是，不能使用 “-” 作为一元运算符。
例如，如果 cards =[1,1,1,1] ，则表达式 “-1 -1 -1 -1” 是 不允许 的。
你不能把数字串在一起
例如，如果 cards =[1,2,1,2] ，则表达式 “12 + 12” 无效。
如果可以得到这样的表达式，其计算结果为 24 ，则返回 true ，否则返回 false 。

<details>

```c++
class Solution {
    constexpr static auto EPS=1e-6;
    const array<function<double(double,double)>,6> ops={
        plus{},
        minus{},
        multiplies{},
        divides{},
        [](auto a,auto b){return b-a;},
        [](auto a,auto b){return b/a;}
    };
public:
    bool judgePoint24(auto&& cards) {
        if(cards.size()==1){
            return abs(cards.front()-24)<EPS;
        }
        for(int i=0;i<cards.size();i++){
            for(int j=i+1;j<cards.size();j++){
                vector<double> tmp(cards.begin(),cards.end());
                auto a=cards[i],b=cards[j];
                for(tmp.erase(tmp.begin()+j);auto &&op:ops){
                    tmp[i]=op(a,b);
                    if(judgePoint24(tmp)) return true;
                }
            }
        }
        return false;
    }
};
```
