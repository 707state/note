# 204 计数质数

给定整数 n ，返回 所有小于非负整数 n 的质数的数量 。

<details>

```cpp
class Solution {
public:
    int countPrimes(int n) {
        vector<int> isPrime(n,1);
        int ans=0;
        for(int i=2;i<n;i++){
            if(isPrime[i]){
                ans+=1;
                if((long long)i*i<n){
                    for(int j=i*i;j<n;j+=i){
                        isPrime[j]=0;
                    }
                }
            }
        }
        return ans;
    }
};
```
